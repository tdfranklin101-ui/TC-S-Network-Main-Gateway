  <script>
    (async function loadSolar() {
      const URL = "https://solarreserveretracker.tdfranklin101.repl.co/solar/reserve";
      const el = (id) => document.getElementById(id);
      try {
        const res = await fetch(URL, { cache: "no-store" });
        if (!res.ok) throw new Error("HTTP " + res.status);
        const data = await res.json();
        const n = (x) => (typeof x === "number" ? x : Number(x || 0));
        const fmt = (x) => n(x).toLocaleString();

        el("w-date").textContent = data.date || "—";
        el("w-kwh").textContent = data.renewables_kwh ? `${fmt(data.renewables_kwh)} kWh` : "—";
        el("w-solar").textContent = data.equivalent_solar ? `${fmt(data.equivalent_solar)} S` : "—";
        el("w-delta").textContent = data.delta_percent !== undefined ? `${data.delta_percent}%` : "—";
        el("w-cumulative").textContent = data.cumulative_solar ? `${fmt(data.cumulative_solar)} S` : "—";
        el("w-updated").textContent = new Date().toLocaleString();
      } catch (e) {
        console.error("Solar Reserve fetch error:", e);
        ["w-date","w-kwh","w-solar","w-delta","w-cumulative","w-updated"].forEach(id=>{el(id).textContent="—"});
      }
    })();
  </script>
