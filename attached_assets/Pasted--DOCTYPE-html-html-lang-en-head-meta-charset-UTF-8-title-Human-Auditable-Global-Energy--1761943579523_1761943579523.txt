<script>
async function loadData(){
  const res = await fetch('/auditlog');
  const data = await res.json();

  // --- compute summaries ---
  const totals = {};
  let globalKwh = 0, globalSolar = 0;
  data.forEach(r=>{
    if(!totals[r.category]) totals[r.category]={kwh:0,solar:0};
    totals[r.category].kwh += +r.kwh;
    totals[r.category].solar += +r.solar_units;
    globalKwh += +r.kwh; globalSolar += +r.solar_units;
  });

  // --- summary boxes ---
  const grid=document.getElementById('summaryGrid');
  Object.keys(totals).forEach(cat=>{
    const div=document.createElement('div');
    div.className='summary-item';
    div.innerHTML=`<h3>${cat}</h3>
      <p>${totals[cat].kwh.toLocaleString(undefined,{maximumFractionDigits:0})} kWh</p>
      <p>${totals[cat].solar.toFixed(2)} Solar Units</p>`;
    grid.appendChild(div);
  });
  document.getElementById('globalTotals').innerText =
    `Total Global Consumption: ${globalKwh.toLocaleString(undefined,{maximumFractionDigits:0})} kWh (${globalSolar.toFixed(2)} Solar Units)`;

  // --- chart ---
  const days=[...new Set(data.map(x=>x.day))];
  const cats=[...new Set(data.map(x=>x.category))];
  const colors=['#00ffe0','#ffa500','#ff007f','#00ff88','#66aaff','#ff4444','#cccc00'];
  const datasets=cats.map((c,i)=>({
    label:c,
    data:days.map(d=>{
      const f=data.find(x=>x.category===c && x.day===d);
      return f?f.kwh/1e9:0; // display GWh
    }),
    borderColor:colors[i%colors.length],
    fill:false,
    tension:0.3
  }));
  new Chart(document.getElementById('energyChart'),{
    type:'line',
    data:{labels:days,datasets},
    options:{
      plugins:{title:{display:true,text:'Daily Energy Use by Category (GWh)',color:'#fff'}},
      scales:{x:{ticks:{color:'#ccc'}},y:{ticks:{color:'#ccc'}}},
      responsive:true, maintainAspectRatio:false
    }
  });

  // --- table ---
  const tbody=document.querySelector('#detailTable tbody');
  data.slice(-100).reverse().forEach(r=>{
    const tr=document.createElement('tr');
    tr.innerHTML=`<td>${r.day}</td>
      <td>${r.category}</td>
      <td>${(+r.kwh).toLocaleString()}</td>
      <td>${(+r.solar_units).toFixed(3)}</td>
      <td>${r.source}</td>
      <td>${r.verification_level}</td>
      <td style="font-family:monospace;font-size:11px;">${r.data_hash?.slice(0,16)}‚Ä¶</td>
      <td>${r.notes||''}</td>`;
    tbody.appendChild(tr);
  });

  // --- export button ---
  document.getElementById('exportBtn').addEventListener('click',()=>{
    const blob=new Blob([JSON.stringify(data,null,2)],{type:'application/json'});
    const url=URL.createObjectURL(blob);
    const a=document.createElement('a');
    a.href=url; a.download='solar_energy_audit_data.json'; a.click();
    URL.revokeObjectURL(url);
  });
}
loadData();
