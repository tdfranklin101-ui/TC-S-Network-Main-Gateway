      async function loadRegion(region) {
        const endpoint = `https://solar-reserve-tracker-tdfranklin101.replit.app/api/solar/reserve?region=${region}`;
        document.getElementById("solar-output").textContent = "Fetching data...";
        try {
          const res = await fetch(endpoint);
          if (!res.ok) throw new Error(`Server returned ${res.status}`);
          const data = await res.json();
          document.getElementById("solar-output").textContent = JSON.stringify(data, null, 2);
        } catch (err) {
          document.getElementById("solar-output").textContent = `Error: ${err.message}`;
        }
      }
